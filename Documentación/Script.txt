CREATE TABLE PLAYER(
	playerID SERIAL PRIMARY KEY NOT NULL,
	nickname VARCHAR(20) UNIQUE NOT NULL,
);

CREATE TABLE SCORE(
	scoreID SERIAL PRIMARY KEY NOT NULL,
	score INT NOT NULL,
	playerID INT NOT NULL,
	CONSTRAINT fk_playerID FOREIGN KEY (playerID) 
	REFERENCES player(playerID)
);
